"use strict";

define('test', ['ss'], function(ss) {
  var $global = this;

  // ExpressionTests.Foo

  function Foo(i, j) {
  }
  var Foo$ = {
    toString: function() {
      return 'Foo';
    },
    sum: function(i) {
      return 0;
    }
  };


  // ExpressionTests.Bar

  function Bar(i, j, f) {
    Foo.call(this, i, j);
  }
  var Bar$ = {
    sum: function() {
      return Foo.prototype.sum.call(this, 1) + 1;
    },
    toString: function() {
      return Foo.prototype.toString.call(this) + ' -> Bar';
    }
  };


  var ns_ExpressionTests = "ExpressionTests";

  var $exports = ss.module('test', '', null,
    {
      Foo: ss.defineClass(Foo, Foo$, [Number, Number], null, [], ns_ExpressionTests),
      Bar: ss.defineClass(Bar, Bar$, [Number, Number, Foo], Foo, [], ns_ExpressionTests)
    });


  return $exports;
});
